21#[derive(Debug, Clone, PartialEq)]
22enum HeaderReference<'a> {
23    None,
24    FromHeader(&'a NiftiHeader),
25    FromFile(&'a Path),
26}
27
28impl<'a> HeaderReference<'a> {
29    fn to_header(&self) -> Result<NiftiHeader> {
30        match self {
31            HeaderReference::FromHeader(h) => Ok((*h).to_owned()),
32            HeaderReference::FromFile(path) => NiftiHeader::from_file(path),
33            HeaderReference::None => Ok(NiftiHeader {
34                sform_code: 2,
35                ..NiftiHeader::default()
36            }),
37        }
38    }
39}

41/// Options and flags which can be used to configure how a NIfTI image is written.
42#[derive(Debug, Clone, PartialEq)]
43pub struct WriterOptions<'a> {
44    /// Where to write the output image (header and/or data).
45    path: PathBuf,
46    /// If given, it will be used to fill most of the header's fields. Otherwise, a default
47    /// `NiftiHeader` will be built and written. In all cases, the `dim`, `datatype` and `bitpix`
48    /// fields will depend only on `data` parameter passed to the `write` function, not on this
49    /// field. This means that the `datatype` defined in `header_reference` will be ignored.
50    /// Because of this, `scl_slope` will be set to 1.0 and `scl_inter` to 0.0.
51    header_reference: HeaderReference<'a>,
52    /// Whether to write the NIfTI file pair. (nii vs hdr+img)
53    write_header_file: bool,
54    /// The volume will be compressed if `path` ends with ".gz", but it can be overriden with the
55    /// `compress` method. If enabled, the volume will be compressed using the specified compression
56    /// level. Default to `Compression::fast()`.
57    compression: Option<Compression>,
58    /// The header file will only be compressed if the caller specifically asked for a path ending
59    /// with "hdr.gz". Otherwise, only the volume will be compressed (if requested).
60    force_header_compression: bool,
61
62    /// Optional ExtensionSequence
63    extension_sequence: Option<ExtensionSequence>,
64}
65
66impl<'a> WriterOptions<'a> {
67    /// Creates a blank new set of options ready for configuration.
68    pub fn new<P>(path: P) -> WriterOptions<'a>
69    where
70        P: AsRef<Path>,
71    {
72        let mut path = path.as_ref().to_owned();
73        if path.extension().is_none() {
74            let _ = path.set_extension("nii");
75        }
76        let write_header_file = is_hdr_file(&path);
77        let compression = if is_gz_file(&path) {
78            Some(Compression::fast())
79        } else {
80            None
81        };
82        WriterOptions {
83            path,
84            header_reference: HeaderReference::None,
85            write_header_file,
86            compression,
87            force_header_compression: write_header_file && compression.is_some(),
88            extension_sequence: None,
89        }
90    }
91
92    /// Sets the reference header.
93    pub fn reference_header(mut self, header: &'a NiftiHeader) -> Self {
94        self.header_reference = HeaderReference::FromHeader(header);
95        self
96    }
97
98    /// Loads a reference header from a Nifti file.
99    pub fn reference_file<P: 'a>(mut self, path: &'a P) -> Self
100    where
101        P: AsRef<Path>,
102    {
103        self.header_reference = HeaderReference::FromFile(path.as_ref());
104        self
105    }
106
107    /// Whether to write the header and data in distinct files.
108    ///
109    /// Will update the output path accordingly.
110    pub fn write_header_file(mut self, write_header_file: bool) -> Self {
111        if self.write_header_file != write_header_file {
112            self.write_header_file = write_header_file;
113        }
114        self
115    }
116
117    /// Whether to compress the output to gz.
118    ///
119    /// Will update the output path accordingly.
120    pub fn compress(mut self, compress: bool) -> Self {
121        if self.compression.is_none() && compress {
122            self.compression = Some(Compression::fast());
123        } else if self.compression.is_some() && !compress {
124            self.compression = None;
125        }
126        self
127    }
128
129    /// Sets the compression level to use when compressing the output.
130    pub fn compression_level(mut self, compression_level: Compression) -> Self {
131        self.compression = Some(compression_level);
132        self
133    }
134
135    /// Sets an extension sequence for the writer
136    pub fn with_extensions(mut self, extension_sequence: ExtensionSequence) -> Self {
137        self.extension_sequence = Some(extension_sequence);
138        self
139    }

141    /// Write a nifti file (.nii or .nii.gz) from an NdArray of any Pod type
142    pub fn write_nifti_with_type<A, S, D>(
143        &self,
144        data: &ArrayBase<S, D>,
145        datatype: NiftiType,
146    ) -> Result<()>
147    where
148        S: Data<Elem = A>,
149        A: Pod,
150        D: Dimension + RemoveAxis,
151    {
152        // do a basic size check
153        if size_of::<A>() != datatype.size_of() {
154            return Err(crate::error::NiftiError::UnsupportedDataType(datatype));
155        }
156
157        let header = self.prepare_header(data, datatype)?;
158        let (header_path, data_path) = self.output_paths();
159
160        // Need the transpose for fortran ordering used in nifti file format.
161        let data = data.t();
162
163        let header_file = File::create(header_path)?;
164        if header.vox_offset > 0.0 {
165            if let Some(compression_level) = self.compression {
166                let mut writer = ByteOrdered::runtime(
167                    GzEncoder::new(header_file, compression_level),
168                    header.endianness,
169                );
170                write_header(writer.as_mut(), &header)?;
171                write_extensions(writer.as_mut(), self.extension_sequence.as_ref())?;
172                write_data::<_, A, _, _, _, _>(writer.as_mut(), data)?;
173                let _ = writer.into_inner().finish()?;
174            } else {
175                let mut writer =
176                    ByteOrdered::runtime(BufWriter::new(header_file), header.endianness);
177                write_header(writer.as_mut(), &header)?;
178                write_extensions(writer.as_mut(), self.extension_sequence.as_ref())?;
179                write_data::<_, A, _, _, _, _>(writer, data)?;
180            }
181        } else {
182            let data_file = File::create(data_path)?;
183            if let Some(compression_level) = self.compression {
184                let mut writer = ByteOrdered::runtime(
185                    GzEncoder::new(header_file, compression_level),
186                    header.endianness,
187                );
188                write_header(writer.as_mut(), &header)?;
189                write_extensions(writer.as_mut(), self.extension_sequence.as_ref())?;
190                let _ = writer.into_inner().finish()?;
191
192                let mut writer = ByteOrdered::runtime(
193                    GzEncoder::new(data_file, compression_level),
194                    header.endianness,
195                );
196                write_data::<_, A, _, _, _, _>(writer.as_mut(), data)?;
197                let _ = writer.into_inner().finish()?;
198            } else {
199                let mut header_writer =
200                    ByteOrdered::runtime(BufWriter::new(header_file), header.endianness);
201                write_header(header_writer.as_mut(), &header)?;
202                write_extensions(header_writer.as_mut(), self.extension_sequence.as_ref())?;
203                let data_writer =
204                    ByteOrdered::runtime(BufWriter::new(data_file), header.endianness);
205                write_data::<_, A, _, _, _, _>(data_writer, data)?;
206            }
207        }
208
209        Ok(())
210    }

212    /// Write a nifti file (.nii or .nii.gz) from an NdArray of DataElements

231    fn prepare_header<T, D>(
232        &self,
233        data: &ArrayBase<T, D>,
234        datatype: NiftiType,
235    ) -> Result<NiftiHeader>
236    where
237        T: Data,
238        D: Dimension,
239    {
240        let mut vox_offset: f32 = 352.0;
241
242        if let Some(extension_sequence) = self.extension_sequence.as_ref() {
243            vox_offset += extension_sequence.bytes_on_disk() as f32;
244        }
245
246        let mut header = NiftiHeader {
247            dim: *Dim::from_slice(data.shape())?.raw(),
248            sizeof_hdr: 348,
249            datatype: datatype as i16,
250            bitpix: (datatype.size_of() * 8) as i16,
251            vox_offset,
252            scl_inter: 0.0,
253            scl_slope: 1.0,
254            magic: *MAGIC_CODE_NIP1,
255            // All other fields are copied from the requested reference header
256            ..self.header_reference.to_header()?
257        };
258
259        if self.write_header_file {
260            header.vox_offset = 0.0;
261            header.magic = *MAGIC_CODE_NI1;
262        }
263
264        // The only acceptable length is 80. If different, try to set it.
265        header.validate_description()?;
266
267        Ok(header)
268    }
269
270    /// Fix the header path extension in case a change in `write_header_file` or `compression`
271    /// broke it.
272    fn output_paths(&self) -> (PathBuf, PathBuf) {
273        let mut path = self.path.clone();
274        let _ = path.set_extension("");
275        match (self.write_header_file, self.compression.is_some()) {
276            (false, false) => (path.with_extension("nii"), path.with_extension("nii")),
277            (false, true) => (path.with_extension("nii.gz"), path.with_extension("nii.gz")),
278            (true, false) => (path.with_extension("hdr"), path.with_extension("img")),
279            (true, true) => {
280                if self.force_header_compression {
281                    (path.with_extension("hdr.gz"), path.with_extension("img.gz"))
282                } else {
283                    (path.with_extension("hdr"), path.with_extension("img.gz"))
284                }
285            }
286        }
287    }
288}

290fn write_extensions<W, E>(
291    mut writer: ByteOrdered<W, E>,
292    extensions: Option<&ExtensionSequence>,
293) -> Result<()>
294where
295    W: Write,
296    E: Endian,
297{
298    let extensions = match extensions {
299        Some(extensions) => extensions,
300        None => {
301            writer.write_u32(0)?;
302            return Ok(());
303        }
304    };
305
306    if extensions.is_empty() {
307        // Write an extender code of 4 zeros, which for NIFTI means that there are no extensions
308        writer.write_u32(0)?;
309        return Ok(());
310    }
311
312    writer.write_all(extensions.extender().as_bytes())?;
313    for extension in extensions.iter() {
314        writer.write_i32(extension.size())?;
315        writer.write_i32(extension.code())?;
316        writer.write_all(extension.data())?;
317    }
318    Ok(())
319}

321fn write_header<W, E>(mut writer: ByteOrdered<W, E>, header: &NiftiHeader) -> Result<()>
322where
323    W: Write,
324    E: Endian,
325{
326    writer.write_i32(header.sizeof_hdr)?;
327    writer.write_all(&header.data_type)?;
328    writer.write_all(&header.db_name)?;
329    writer.write_i32(header.extents)?;
330    writer.write_i16(header.session_error)?;
331    writer.write_u8(header.regular)?;
332    writer.write_u8(header.dim_info)?;
333    for s in &header.dim {
334        writer.write_u16(*s)?;
335    }
336    writer.write_f32(header.intent_p1)?;
337    writer.write_f32(header.intent_p2)?;
338    writer.write_f32(header.intent_p3)?;
339    writer.write_i16(header.intent_code)?;
340    writer.write_i16(header.datatype)?;
341    writer.write_i16(header.bitpix)?;
342    writer.write_i16(header.slice_start)?;
343    for f in &header.pixdim {
344        writer.write_f32(*f)?;
345    }
346    writer.write_f32(header.vox_offset)?;
347    writer.write_f32(header.scl_slope)?;
348    writer.write_f32(header.scl_inter)?;
349    writer.write_i16(header.slice_end)?;
350    writer.write_u8(header.slice_code)?;
351    writer.write_u8(header.xyzt_units)?;
352    writer.write_f32(header.cal_max)?;
353    writer.write_f32(header.cal_min)?;
354    writer.write_f32(header.slice_duration)?;
355    writer.write_f32(header.toffset)?;
356    writer.write_i32(header.glmax)?;
357    writer.write_i32(header.glmin)?;
358
359    writer.write_all(&header.descrip)?;
360    writer.write_all(&header.aux_file)?;
361    writer.write_i16(header.qform_code)?;
362    writer.write_i16(header.sform_code)?;
363    for f in &[
364        header.quatern_b,
365        header.quatern_c,
366        header.quatern_d,
367        header.quatern_x,
368        header.quatern_y,
369        header.quatern_z,
370    ] {
371        writer.write_f32(*f)?;
372    }
373    for f in header
374        .srow_x
375        .iter()
376        .chain(&header.srow_y)
377        .chain(&header.srow_z)
378    {
379        writer.write_f32(*f)?;
380    }
381    writer.write_all(&header.intent_name)?;
382    writer.write_all(&header.magic)?;
383
384    Ok(())
385}

387/// Write the data in 'f' order.
388///
389/// Like NiBabel, we iterate by "slice" to improve speed and use less memory.
390fn write_data<A, B, S, D, W, E>(mut writer: ByteOrdered<W, E>, data: ArrayBase<S, D>) -> Result<()>
391where
392    S: Data<Elem = A>,
393    A: Pod,
394    D: Dimension + RemoveAxis,
395    W: Write,
396    E: Endian + Copy,
397{
398    // We always write the image by iterating on the first axis, thus
399    //   3D (depth, height, width) => per axial slice
400    //   4D (time, depth, height, width) => per volume
401    // However, there's a problem when dim == (1, ...). We must iterate by the first non-one length
402    // axis, otherwise the image will be written in the wrong ordering. See issue #63.
403    let first_good_axis = data.shape().iter().position(|&d| d > 1).unwrap_or(0);
404    let mut iter = data.axis_iter(Axis(first_good_axis));
405    if let Some(arr_data) = iter.next() {
406        // Keep slice voxels in a separate array to ensure `C` ordering even after `into_shape`.
407        let mut slice = arr_data.to_owned();
408        write_slice::<_, B, _, _, _, _>(writer.as_mut(), slice.view())?;
409        for arr_data in iter {
410            slice.assign(&arr_data);
411            write_slice::<_, B, _, _, _, _>(writer.as_mut(), slice.view())?;
412        }
413    }
414    Ok(())
415}

417fn write_slice<A, B, S, D, W, E>(
418    writer: ByteOrdered<&mut W, E>,
419    data: ArrayBase<S, D>,
420) -> Result<()>
421where
422    S: Data<Elem = A>,
423    A: Clone + Pod,
424    D: Dimension,
425    W: Write,
426    E: Endian,
427{
428    let len = data.len();
429    let arr_data = data.into_shape(len).unwrap();
430    let slice = arr_data.as_slice().unwrap();
431    let bytes = cast_slice(slice);
432    let (writer, endianness) = writer.into_parts();
433    let bytes = adapt_bytes::<B, _>(bytes, endianness);
434    writer.write_all(&bytes)?;
435    Ok(())
