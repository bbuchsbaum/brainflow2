1//! This module contains definitions for the extension and related types.
2//! Extensions are optional data frames sitting before the voxel data.
3//! When present, an extender frame of 4 bytes is also present at the
4//! end of the NIFTI-1 header, with the first byte set to something
5//! other than 0.
6
7use crate::error::{NiftiError, Result};
8use byteordered::{ByteOrdered, Endian};
9use std::io::{ErrorKind as IoErrorKind, Read};
10
11/// Data type for the extender code.
12#[derive(Debug, Default, PartialEq, Clone, Copy)]
13pub struct Extender([u8; 4]);
14
15impl Extender {
16    /// Fetch the extender code from the given source, while expecting it to exist.
17    pub fn from_reader<S: Read>(mut source: S) -> Result<Self> {
18        let mut extension = [0u8; 4];
19        source.read_exact(&mut extension)?;
20        Ok(extension.into())
21    }
22
23    /// Fetch the extender code from the given source, while
24    /// being possible to not be available.
25    /// Returns `None` if the source reaches EoF prematurely.
26    /// Any other I/O error is delegated to a `NiftiError`.
27    pub fn from_reader_optional<S: Read>(mut source: S) -> Result<Option<Self>> {
28        let mut extension = [0u8; 4];
29        match source.read_exact(&mut extension) {
30            Ok(()) => Ok(Some(extension.into())),
31            Err(ref e) if e.kind() == IoErrorKind::UnexpectedEof => Ok(None),
32            Err(e) => Err(NiftiError::from(e)),
33        }
34    }
35
36    /// Whether extensions should exist upon this extender code.
37    pub fn has_extensions(&self) -> bool {
38        self.0[0] != 0
39    }
40
41    /// Get the extender's bytes
42    pub fn as_bytes(&self) -> &[u8; 4] {
43        &self.0
44    }
45}
46
47impl From<[u8; 4]> for Extender {
48    fn from(extender: [u8; 4]) -> Self {
49        Extender(extender)
50    }
51}

53/// Data type for the raw contents of an extension.
54/// Users of this type have to reinterpret the data
55/// to suit their needs.
56#[derive(Debug, PartialEq, Clone)]
57pub struct Extension {
58    esize: i32,
59    ecode: i32,
60    edata: Vec<u8>,
61}
62
63impl Extension {
64    /// Create an extension out of its main components.
65    pub fn new(ecode: i32, edata: Vec<u8>) -> Self {
66        let esize = 8 + edata.len() as i32;
67
68        Extension {
69            esize,
70            ecode,
71            edata,
72        }
73    }
74
75    /// Create a new extension out of a &str
76    pub fn from_str(ecode: i32, edata: &str) -> Self {
77        let esize = 8 + edata.len() as i32;
78        // pad the esize to a multiple of 16
79        let padded_esize = (esize + 15) & !15;
80        let mut edata = edata.as_bytes().to_vec();
81        edata.resize(padded_esize as usize - 8, 0);
82        Extension::new(ecode, edata)
83    }
84
85    /// Obtain the claimed extension raw size (`esize` field).
86    pub fn size(&self) -> i32 {
87        self.esize
88    }
89
90    /// Obtain the extension's code (`ecode` field).
91    pub fn code(&self) -> i32 {
92        self.ecode
93    }
94
95    /// Obtain the extension's data (`edata` field).
96    pub fn data(&self) -> &Vec<u8> {
97        &self.edata
98    }
99
100    /// Take the extension's raw data, discarding the rest.
101    pub fn into_data(self) -> Vec<u8> {
102        self.edata
103    }
104}

106/// Data type for aggregating the extender code and
107/// all extensions.
108#[derive(Debug, PartialEq, Clone)]
109pub struct ExtensionSequence {
110    extender: Extender,
111    extensions: Vec<Extension>,
112}
113
114impl IntoIterator for ExtensionSequence {
115    type Item = Extension;
116    type IntoIter = ::std::vec::IntoIter<Self::Item>;
117
118    fn into_iter(self) -> Self::IntoIter {
119        self.extensions.into_iter()
120    }
121}
122
123impl<'a> IntoIterator for &'a ExtensionSequence {
124    type Item = &'a Extension;
125    type IntoIter = ::std::slice::Iter<'a, Extension>;
126
127    fn into_iter(self) -> Self::IntoIter {
128        self.iter()
129    }
130}
131
132impl ExtensionSequence {
133    /// Provide a public constructor
134    pub fn new(extender: Extender, extensions: Vec<Extension>) -> Self {
135        ExtensionSequence {
136            extender,
137            extensions,
138        }
139    }
140
141    /// Read a sequence of extensions from a source, up until `len` bytes.
142    pub fn from_reader<S, E>(
143        extender: Extender,
144        mut source: ByteOrdered<S, E>,
145        len: usize,
146    ) -> Result<Self>
147    where
148        S: Read,
149        E: Endian,
150    {
151        let mut extensions = Vec::new();
152        if extender.has_extensions() {
153            let mut offset = 0;
154            while offset < len {
155                let esize = source.read_i32()?;
156                let ecode = source.read_i32()?;
157
158                let data_size = (esize as usize).saturating_sub(8);
159                let mut edata = Vec::new();
160                edata
161                    .try_reserve_exact(data_size)
162                    .map_err(|e| NiftiError::ReserveExtended(data_size, e))?;
163                let nb_bytes_written = (&mut source)
164                    .take(data_size as u64)
165                    .read_to_end(&mut edata)?;
166
167                if nb_bytes_written != data_size {
168                    return Err(NiftiError::IncompatibleLength(nb_bytes_written, data_size));
169                }
170
171                extensions.push(Extension::new(ecode, edata));
172                offset += esize as usize;
173            }
174        }
175
176        Ok(ExtensionSequence {
177            extender,
178            extensions,
179        })
180    }
181
182    /// Obtain an iterator to the extensions.
183    pub fn iter(&self) -> ::std::slice::Iter<Extension> {
184        self.extensions.iter()
185    }
186
187    /// Whether the sequence of extensions is empty.
188    pub fn is_empty(&self) -> bool {
189        self.extensions.is_empty()
190    }
191
192    /// Obtain the number of extensions available.
193    pub fn len(&self) -> usize {
194        self.extensions.len()
195    }
196
197    /// Return the number of bytes the extensions take on disk
198    pub fn bytes_on_disk(&self) -> usize {
199        self.extensions
200            .iter()
201            .map(|e| e.size() as usize)
202            .sum::<usize>()
203    }
204    /// Get the extender code from this extension sequence.
205    pub fn extender(&self) -> Extender {
206        self.extender
207    }
208}
